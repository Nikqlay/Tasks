Допишите функцию rgb(). В качестве аргументов она принимает десятичные значения RGB. Функция должна вернуть значение цвета в шестнадцатеричном формате. Допустимые значения RGB в десятичном формате равны 0-255. Если числа выходят за указанный диапазон, то нужно округлить их до ближайшего возможного числа. Ниже приведены примеры.

Например:
Вход					Выход
rgb(255, 255, 255)		FFFFFF

function diapazon ($r){
        if ($r <= 0) {
            $r = 0;
        }
        elseif ($r >= 255) {
            $r = 255;
        }
$r = strtoupper(dechex($r));
if ((strlen ($r)<2)) {
	$r = (str_pad((0), 2, $r, STR_PAD_LEFT));
}
return $r;
}

function rgb($r, $g, $b){
$r = diapazon ($r);
$g = diapazon ($g);
$b = diapazon ($b);
    
$s = $r.$g.$b;
return $s;
}
