Нужно написать функцию, которая возвращает количество конечных нулей в факториале переданного числа.

N! = 1 * 2 * 3 * 4 ... N

Например:
Вход	Выход
zeros(12) 2

function zeros( $n){
$i=0;
$s = gmp_fact($n);
$s = gmp_strval($s);

$i=0;
foreach ($s as $key) {
	if ($key == 0) {
	$i++;
	}
	else {
	$i=0;
	}
}
return ($i);
}
